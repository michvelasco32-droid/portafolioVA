Este programa crea un visualizador interactivo de im√°genes que permite hacer zoom in y zoom out en tiempo real. Utilizando la biblioteca OpenCV de Python, puedes ampliar o reducir una imagen mediante comandos de teclado, perfecto para inspeccionar detalles o tener una vista general de la imagen. 

def ampliar():
    import cv2
    # Cargo la imagen en una variable
    img_original = cv2.imread("myhero.jfif")
    # Creando variable para el valor del zoom original
    zoom = 1.0

    while True:
        img = cv2.resize(img_original, None, fx=zoom, fy=zoom)
        cv2.imshow("imagen nueva", img)
        tecla = cv2.waitKey(0)
        
        if tecla == ord('+'):
            zoom = zoom * 1.2
        elif tecla == ord("-"):
            zoom = zoom - 0.2
        elif tecla == ord('o'):
            zoom = 1.0
        elif tecla == ord('x'):
            break
